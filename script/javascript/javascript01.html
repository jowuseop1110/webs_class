01.변수 : 데이터 저장

{
    var x = 100;
    var y = 200;
    var z = "jvascript";
    // 변수 x에 숫자 100을 저장
    // 변수 y에 숫자 200을 저장
    // 변수 z에 문자 javascript를 저장(문자를 입력할떄는 ""를 입력해야 한다)  
                                                        
    document.write("*** 01. 변수 ***");
    document.write(x);
    document.write(y);
    document.write(z);
}

02.변수 : 데이터 저장 + 데이터 변경

{
    let x = 100;
    let y = 200;
    let z = "javascript";
    // 데이터 저장소
    x = 300;
    // 100 -> 300으로 데이터 변경
    y = 400;
    // 200 -> 400으로 데이터 변경
    z = "jquery";
    // "javascript" -> "jquery"로 데이터 변경

    document.write("*** 02.변수 ***");
    document.write(x);
    document.write(y);
    document.write(z);
}

03.변수 : 데이터 저장 + 데이터 변경 + 데이터 추가

{
    let x = 100;
    let y = 200;
    let z = "javascript";
    x += 300;
    // 기존의 x값(100)에다가 300을 + 해주겠다
    y += 400;
    z += ", jquery"

    document.write("*** 03.변수 ***");
    document.write(x);
    document.write(y);
    document.write(z);
}
