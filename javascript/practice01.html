1. 1부터 100까지 출력하기

1.1 while문을 이용해서 1부터 100까지 출력하는 함수를 작성해주세요.

{
    let i = 1;
    while(i<=100){
        document.write(i);
        i++;
    }
}

1.2 for문 이용해서 1부터 100까지 출력하는 함수를 작성해주세요

{
    for( let i = 1; i <= 100; i++){
        document.write(i);
    }
}

1.3 for문 이용해서 1부터 100까지 출력하는 함수를 작성해주세요

조건1. 짝수만 출력하세요

조건2. 홀수만 출력하세요

{
    // 짝수
    for (let i = 1; i <= 100; i++) {

        if (i % 2 == 0) {
            document.write(i);
        }
    }
    // 홀수
    for (let i = 1; i <= 100; i++) {

        if (i % 2 == 1) {
            document.write(i);
        }
    }
    // 짝수 다른 방법1
    for (let i = 2; i <= 100; i++) {
        if( i % 2 ==0 )
        document.write(i);
    }
    // 짝수 다른 방법2
    for (let i = 2; i <= 100; i+=2) {
        document.write(i);               
    }
    // 홀수 다른 방법1
    for (let i = 1; i <= 100; i+=2) {
        document.write(i);               
    }
}

1.4 for문 이용해서 1부터 100까지 합을 출력하세요.

{
    let sum = 0;
    for(let i=1; i<=100; i++){
        sum = sum + i ;   
    }
    document.write(sum);
}
